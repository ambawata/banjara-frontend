import express from "express"; import dotenv from "dotenv"; import cors from "cors"; import OpenAI from "openai"; import multer from "multer"; import sharp from "sharp"; import { MongoClient, ObjectId, GridFSBucket } from "mongodb"; import { Readable } from "stream"; dotenv.config(); const app = express(); app.use(cors()); app.use(express.json()); // API key check ko yahan add kiya gaya hai if (!process.env.OPENAI_API_KEY) { console.error("âŒ The OPENAI_API_KEY environment variable is missing or invalid."); // Optionally, exit the application if the key is critical // process.exit(1); } const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY }); const client = new MongoClient(process.env.MONGODB_URI); let db, chats, bucket; async function connectToMongo() { try { await client.connect(); db = client.db(); chats = db.collection("chats"); bucket = new GridFSBucket(db, { bucketName: "uploads" }); console.log("âœ… Connected to MongoDB"); } catch (err) { console.error("âŒ MongoDB connection error:", err); } } connectToMongo(); // Home route app.get("/", (req, res) => { res.send("ðŸ‘‹ Welcome to BanjaraGPT API!
